#include <Arduino.h>

#define LED_PIN LED_BUILTIN

#define SERVICE_UUID        "4fafc202-1fb5-459e-8fcc-c5c9c331914b"
#define CHARACTERISTIC_UUID "cec5483e-36e1-4688-b7f5-ea07361b26a8"

volatile int isr_num = 0;

//motor driver pins=============
#define LEFT_IN1 33
#define LEFT_IN2 25
#define RIGHT_IN1 26
#define RIGHT_IN2 27

#define LEFT_EN 12
#define RIGHT_EN 13

bool on_state = false;

//ir pins
#define SR2 36
#define SR1 39
#define S0 34
#define SL1 35
#define SL2 32

//=======================================

BLECharWrapper* dataChar = nullptr;
BLECharWrapper* rfidChar = nullptr;
uint32_t value = 0;
bool ledState=false;

//ISR====================
void IRAM_ATTR isr_s0(){
    isr_num = 1;
}

void IRAM_ATTR isr_sl2(){
    isr_num = 2;
}

void IRAM_ATTR isr_sr2(){
    isr_num = 3;
}



void setup() {
    Serial.begin(115200);
  pinMode(LED_PIN, OUTPUT);
    pinMode(LEFT_IN1, OUTPUT);
  pinMode(LEFT_IN2, OUTPUT);
  pinMode(RIGHT_IN1, OUTPUT);
  pinMode(RIGHT_IN2, OUTPUT);

  pinMode(LEFT_EN, OUTPUT);
  pinMode(RIGHT_EN, OUTPUT);

  digitalWrite(LEFT_EN, LOW);
  digitalWrite(RIGHT_EN, LOW);

  //ir pins setup==============
    pinMode(S0, INPUT);
  pinMode(SL1, INPUT);
  pinMode(SL2, INPUT);
  pinMode(SR1, INPUT);
  pinMode(SR2, INPUT);
//====interrupt setup==========
    attachInterrupt(digitalPinToInterrupt(S0), isr_s0, RISING);
    attachInterrupt(digitalPinToInterrupt(SL2), isr_sl2, RISING);
    attachInterrupt(digitalPinToInterrupt(SR2), isr_sr2, RISING);
//=======================================

  BLEManager& ble = BLEManager::getInstance();
    ble.init("my_esp_test");


    // Set connection callbacks
    ble.onConnect([]() {
        Serial.println("Client connected!");
    });

    ble.onDisconnect([]() {
        Serial.println("Client disconnected, restarting advertising...");
        BLEManager::getInstance().startAdvertising();
    });

    // Create service
    BLEServiceWrapper* service = ble.createService(SERVICE_UUID);

    // Create characteristic with builder pattern
    dataChar = service->addCharacteristic(CHARACTERISTIC_UUID)
        .withRead()
        .withWrite()
        .withNotify()
        .onWrite([](uint8_t* data, size_t len) {
            if (len == 0 || len > 31) return;

            char buf[32];
            memcpy(buf, data, len);
            buf[len] = '\0';
            Serial.print("Received: ");
            Serial.println(buf);

            if(strcmp(buf, "1")==0){
              on_state = !on_state;

              if(on_state == true){
              digitalWrite(LEFT_EN, HIGH);
              digitalWrite(RIGHT_EN, HIGH);

              digitalWrite(LEFT_IN1, HIGH);
              digitalWrite(LEFT_IN2, LOW);

              digitalWrite(RIGHT_IN1, HIGH);
              digitalWrite(RIGHT_IN2, LOW);
             
              }
              else{
              digitalWrite(LEFT_EN, LOW);
              digitalWrite(RIGHT_EN, LOW);
              }
        }
    })
        .build();

            service->start();
    ble.startAdvertising();

    Serial.println("BLE Server ready. Waiting for connections...");


}
